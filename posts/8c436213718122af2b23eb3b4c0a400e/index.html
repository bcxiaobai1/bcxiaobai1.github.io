 var_dump($username,$content);// 检验传来的数据是否有误

数据库界面（上述代码中7.php为下面代码的文件名）

<?php
$dsn='mysql:dbname=test;host=127.0.0.1';
$pdo=new PDO($dsn,'root','root');

function write($pdo,$sql){
$sth=$pdo->prepare($sql);
return $sth->execute();
}

function read($pdo,$sql){
$sth=$pdo->prepare($sql);
$sth->execute();
$rows=$sth->fetchAll();
return $rows;
}
?>
