package ai.hy.csdn.tools;

import cn.hutool.core.util.IdUtil;
import com.google.common.base.Joiner;
import com.google.common.base.Splitter;

import java.io.File;
import java.text.MessageFormat;
import java.util.Arrays;
import java.util.List;

/** @Author 剑客阿良_ALiang @Date 2021/11/18 10:05 @Description: 模糊工具 */
public class BlurTool {
  /**
   * 模糊图片
   *
   * @param imagePath MP4视频路径
   * @param outputDir 临时目录
   * @return 图片地址
   * @throws Exception 异常
   */
  public static String blurImage(String imagePath, String outputDir, Integer level)
      throws Exception {
    List<String> paths = Splitter.on(".").splitToList(imagePath);
    String ext = paths.get(paths.size() - 1);
    if (!Arrays.asList("png", "jpg").contains(ext)) {
      throw new Exception("format error");
    }
    String resultPath =
        Joiner.on(File.separator).join(Arrays.asList(outputDir, IdUtil.simpleUUID() + "." + ext));
    ProcessBuilder builder =
        new ProcessBuilder(
            "C:\ffpg\bin\ffmpeg.exe",
            "-i",
            imagePath,
            "-filter_complex",
            MessageFormat.format("boxblur={0}:1:cr=0:ar=0", String.valueOf(level)),
            resultPath);
    builder.inheritIO().start().waitFor();
    return resultPath;
  }

  /**
   * 模糊视频
   *
   * @param videoPath MP4视频路径
   * @param outputDir 临时目录
   * @return 图片地址
   * @throws Exception 异常
   */
  public static String blurVideo(String videoPath, String outputDir, Integer level)
      throws Exception {
    List<String> paths = Splitter.on(".").splitToList(videoPath);
    String ext = paths.get(paths.size() - 1);
    if (!Arrays.asList("mp4", "flv", "avi").contains(ext)) {
      throw new Exception("format error");
    }
    String resultPath =
        Joiner.on(File.separator).join(Arrays.asList(outputDir, IdUtil.simpleUUID() + "." + ext));
    ProcessBuilder builder =
        new ProcessBuilder(
            "C:\ffpg\bin\ffmpeg.exe",
            "-i",
            videoPath,
            "-filter_complex",
            MessageFormat.format("boxblur={0}:1:cr=0:ar=0", String.valueOf(level)),
            resultPath);
    builder.inheritIO().start().waitFor();
    return resultPath;
  }
}

代码说明

1、blurImage与blurVideo分别为对图片和对视频的处理方法，参数均为：原始地址、输出目录、以及模糊等级。

2、ffmpeg需要安装，在方法中使用本地的安装好的ffmpeg执行文件地址，如何安装可以参考我的另一篇文章：windows ffmpeg安装部署_阿良的博客-CSDN博客

3、模糊等级越高，模糊效果越明显，默认可以设置为10。

4、防止文件重名，使用了uuid作为文件名。
