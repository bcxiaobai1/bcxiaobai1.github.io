6．编写计算最小公倍数的函数，试由主函数输入两个正整数a和b调用之。计算最小公

倍数的公式为：

lcm(u,v)=u*v/gcd(u,v) (u,v≥0)

其中，gcd(u,v)是u、v的最大公约数。lcm(u，v)是u、v的最小公倍数。

#include <stdio.h>

#include <stdlib.h>

int gcd(int u,int v);

int lcm(int u,int v);

int main()

{

    int a,b,x;

    printf("please input 'a' and 'b':");

    scanf("%d %d",&a,&b);

    x=lcm(a,b);

    printf("the lcm:%d",x);

}

int gcd(int u,int v)

{

int c,i,g;

c=u<v? u:v;

for(i=1;i<=c;i++)

{

if(u%i==0&&v%i==0)

{

g=i;

}

}

return g;

}

int lcm(int u,int v)

{

return (u*v)/gcd(u,v);

}
