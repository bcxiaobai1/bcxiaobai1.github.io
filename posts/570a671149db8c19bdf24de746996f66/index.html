# 创建平移矩阵  x方向上平移100，y方向上平移50

5.图像的旋转

row,col = rain.shape[:2]

# 生成旋转矩阵 getRotationMatrix2D(旋转中心坐标，旋转角度，缩放比例)
M = cv.getRotationMatrix2D((row/2,col/2),45,1)

# cv.warpAffine(要旋转的图形,旋转矩阵,背景画布大小)
res = cv.warpAffine(rain,M,(col,row))
