void zy()
{
	short _n = 0;
	char s;
	system("cls");
	if (ydjf > 0)
	{
		cout << "运动积分：" << ydjf << "n";
		_n++;
	}

	{
		cout << "跑步最高记录：" << pbjl / 1000.0 << "秒n";
		_n++;
	}
	for (long long i = 1; i <= 8 - _n; i++)
		cout << "n";
	cout << "                                         运动会";
	cout << "nnn                                         0.退出";
	cout << "nnn                                         1.跑步";
	s = _getch();
	while (s != '0' && s != '1')s = _getch();
	switch (s)
	{
	case '0':
		cx_flag = 0;
		return;
	case '1':
		paobu();
		break;
	}
	return;
}
