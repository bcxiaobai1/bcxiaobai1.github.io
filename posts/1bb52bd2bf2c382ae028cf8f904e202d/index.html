 //密码隐藏化 
 void get_password(char *pswd, unsigned maxlen) {
    int index = 0;
    char buff = '';

    while ((buff = getch()) != 'r') {
        if (buff == 'b' && index != 0) {
            index--;
            printf("b b");
        } else if (index < maxlen - 1 && buff != 'b') {
            pswd[index++] = buff;
            putchar('*');
        }
    }
    pswd[index] = '';
    printf("n");
}
