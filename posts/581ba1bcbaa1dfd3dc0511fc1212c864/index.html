# 单词搜索2
"""
给定一个 m x n 二维字符网格 board 和一个单词（字符串）列表 words，返回所有二维网格上的单词 。
单词必须按照字母顺序，通过 相邻的单元格 内的字母构成，其中“相邻”单元格是那些水平相邻或垂直相邻的单元格。
同一个单元格内的字母在一个单词中不允许被重复使用。
"""


# 超时解法
class Solution1:
    def findWords(self, board, words):
        hang = len(board)
        lie = len(board[0])
        flag_array = []
        result_lst = []
        for i in range(hang):
            flag_array.append([0] * lie)

        def dfs(h, l, w, rw):
            if w != rw[:len(w)]:
                return
            if w == rw and w not in result_lst:
                result_lst.append(w)
                return
            if l + 1 < lie and flag_array[h][l + 1] == 0:
                flag_array[h][l + 1] = 1
                dfs(h, l + 1, w + board[h][l + 1], rw)
                flag_array[h][l + 1] = 0
            if l - 1 > -1 and flag_array[h][l - 1] == 0:
                flag_array[h][l - 1] = 1
                dfs(h, l - 1, w + board[h][l - 1], rw)
                flag_array[h][l - 1] = 0
            if h + 1 < hang and flag_array[h + 1][l] == 0:
                flag_array[h + 1][l] = 1
                dfs(h + 1, l, w + board[h + 1][l], rw)
                flag_array[h + 1][l] = 0
            if h - 1 > -1 and flag_array[h - 1][l] == 0:
                flag_array[h - 1][l] = 1
                dfs(h - 1, l, w + board[h - 1][l], rw)
                flag_array[h - 1][l] = 0

        for word in words:
            for h in range(hang):
                for l in range(lie):
                    if board[h][l] == word[0]:
                        flag_array[h][l] = 1
                        dfs(h, l, word[0], word)
                        flag_array[h][l] = 0
        return result_lst

class Solution:
    def findWords(self, board, words):
        m, n = len(board), len(board[0])

        root = {}
        for w in words:
            p = root
            for c in w:
                if c not in p:
                    p[c] = {}
                p = p[c]
            p['finish'] = w
        # print(root) root构建字典树 真的牛
        # {'o': {'a': {'t': {'finish': 'oat', 'h': {'finish': 'oath'}}}},
        # 'p': {'e': {'a': {'finish': 'pea'}}},
        # 'e': {'a': {'t': {'finish': 'eat'}}},
        # 'r': {'a': {'i': {'n': {'finish': 'rain'}}}}}

        def dfs(start, p):
            i, j = start[0], start[1]
            c = board[i][j]

            last = p[c].pop('finish', False)
            if last:
                res.append(last)

            board[i][j] = '#'
            for x, y in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                ni, nj = i + x, j + y
                if 0 <= ni < m and 0 <= nj < n and board[ni][nj] in p[c]:
                    dfs((ni, nj), p[c])
            board[i][j] = c

            if not p[c]:
                p.pop(c)

        res = []
        for i in range(m):
            for j in range(n):
                if board[i][j] in root:
                    dfs((i, j), root)
        return res
