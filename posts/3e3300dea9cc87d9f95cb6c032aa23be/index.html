?‍如多个英文字母所组成的一个英文单词。
?字符串中可以包含任意字符，这些字符必须包含在一对双引号“”之内。
?‍例如“abc”。
?在Java中定义了String和StringBuffer两个类来封装字符串，并提供一系列操作字符串的方法，它们位于java.lang包中，因此不需要导报就可以直接使用。

package StringAndStringBuffer;

import java.util.Iterator;
import java.util.Scanner;

public class Demo1 {
	public static void main(String[] args) {
//		indexof()方法返回
		String stuName1 = "王小王";
		String stuName2 = "Mike";
		String stuName3 = "王小王";
		String stuName4 = "王小二";
		
		String fileName1 = "eclipse.exe";
		String fileName2 = "";
		
//		脏话字符串数组
		String [] errs = new String[] {"操","日","SB","sb","妈的"};
		
		String[] stuArrary = new String[4];
		stuArrary[0] = stuName1;
		stuArrary[1] = stuName2;
		stuArrary[2] = stuName3;
		stuArrary[3] = stuName4;

		
		
//		获取字符串“王小王”中第一个出现字符串王的下标
		System.out.println(stuName1.indexOf("王"));
		
//		获取字符串“王小王”中最后一个出现字符串王的下标
		System.out.println(stuName1.lastIndexOf("王"));
		
//		chatAt返回字符串中index位置上的字符
		System.out.println(stuName1.charAt(0));
		 
//		endWith()判断该字符串是否已指定的字符串结尾
		if(fileName1.endsWith("exe")) {
			System.err.println(fileName1 + "是可执行文件");
		}
		
//		length()返回字符串的长度
		System.out.println(fileName1 +"的长度是:"+ fileName1.length());
		
		
//		equals方法的使用(字符串和指定字符串的比较)
		if(stuName1.equals(stuName3)) {
			System.out.println(stuName1 + "和" + stuName3 +"是同名同姓的");
		}
		
//		isEmpty方法当且仅当字符串长度为0时返回true
		System.out.println(fileName2.isEmpty());
		
		
//		在学生名单中找出姓王的同学并打印出来,使用到了startsWith()方法
		for (int i = 0; i < stuArrary.length; i++) {
			if(stuArrary[i].startsWith("王")) {
				System.out.println(stuArrary[i]);
			}
		}	
		
//		在聊天时候,常会出现一些不文明的话,需要使用*代替
		Scanner sc = new Scanner(System.in);
		String says = sc.nextLine();
		String s1 = null ;
		
//		循环遍历每一个脏话
		for (int i = 0; i < errs.length; i++) {
//			如果用户说的话里面包含了脏话
			if(says.contains(errs[i])) {
				if(errs[i].length() == 1 ) {
					 s1= says.replace(errs[i], "*");
				}
			}
		}
		System.out.println(s1);
		
		
		
//		toLowerCase方法将大写转换称小写
		System.out.println("ABC".toLowerCase());
		
//		toUpperCase方法将小写转换称大写
		System.out.println("abc".toUpperCase());
		
		
		//将数字转换字符串
		String s = "";
		String str = s.valueOf(123);
		System.out.println(str);

//		toCharArrary()方法将字符串转换成字符数组
		String str1 ="abc";
		char [] chars = str1.toCharArray();
		
		
//		substring方法截取一段返回字符串
		String str2 = "abc";
		String str3 = str2.substring(0,3);
		System.out.println(str3);
		
		
//		trim()方法去除空格
		String str5 = "             abc ";
		System.out.println("没有去除空格的字符串:" +"_"+ str5);
		System.out.println("去除空格后的字符串:"  +"_"+str5.trim());
	}		
	}
