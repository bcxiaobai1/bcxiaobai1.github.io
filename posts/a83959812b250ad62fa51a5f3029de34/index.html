1.以下程序段（ AD）的功能是：输入一批整数，用负数作为输入的结束标志，统计其中大于85的数据个数。

A.

int count = 0, score;
scanf ("%d", &score);
while(score >= 0){
       if(score > 85){
           count++;
       }
       scanf ("%d", &score);
}
printf("%dn", count);

B.

int count = 0, score;
scanf ("%d", &score);
while(score >= 0){
     scanf ("%d", &score);
     if(score > 85){
         count++;
     }
}
printf("%dn", count);

C.scanf位置不对，放到了while里面。如果有负数它不会结束。

int count = 0, score;
while(score >= 0){
       scanf ("%d", &score);
       if(score > 85){
           count++;
       }
}
printf("%dn", count);

D.

int count = 0, score;
while(1){
        scanf ("%d", &score);
        if(score < 0) break;
        if(score > 85){
            count++;
        }
}
printf("%dn", count);
