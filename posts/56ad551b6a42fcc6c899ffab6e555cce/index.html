3.java功能文件

注意，上述代码在对字符串进行非空判断的时候使用了
TextUtils.isEmpty() 方法，这是一个非常好用的方法，它可以一次
性进行两种空值的判断。当传入的字符串等于null 或者等于空字符串
的时候，这个方法都会返回true ，从而使得我们不需要先单独判断这
两种空值再使用逻辑运算符连接起来了。
现在重新运行一下程序，刚才保存的Content字符串肯定会被填充到
EditText中，然后编写一点其他的内容，比如在EditText中输入Hello，
接着按下Back键退出程序，再重新启动程序，这时刚才输入的内容并不
会丢失，而是还原到了EditText中，如图6.6所示。

package com.example.filepersistencetest;

import androidx.appcompat.app.AppCompatActivity;

import android.app.AlarmManager;
import android.app.PendingIntent;
import android.content.Context;
import android.content.Intent;
import android.os.Bundle;
import android.text.TextUtils;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;
import android.widget.Toast;

import java.io.BufferedReader;
import java.io.BufferedWriter;
import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.io.IOException;
import java.io.InputStreamReader;
import java.io.OutputStreamWriter;

public class MainActivity extends AppCompatActivity implements View.OnClickListener {

    private EditText edit;//文件编辑框
    private Button button;//保存按钮
    private Button readData;//读取按钮

    @Override
    protected void onCreate(Bundle savedInstanceState){
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        //id 绑定
        edit = (EditText) findViewById(R.id.edit);
        button = (Button) findViewById(R.id.button);
        button.setOnClickListener(this);
        readData = (Button) findViewById(R.id.readData);
        readData.setOnClickListener(this);

        //一开始初始化，填写读取到的文件值
        String inputText = load();
        if (!TextUtils.isEmpty(inputText)) {//不为空就填写
            edit.setText(inputText);
            edit.setSelection(inputText.length());
            Toast.makeText(this, "重启应用读取数据成功"+inputText, Toast.LENGTH_SHORT).show();//吐司提示更新成功
        }
    }

    //按钮点击监听
    public void onClick(View view) {
        int id = view.getId();
        switch (id){
            case R.id.button:
                String inputText = edit.getText().toString();
                save(inputText);//在点击手机的back退出后保存
                break;
            case R.id.readData:
                String readMessage=load();
                if (!TextUtils.isEmpty(readMessage)) {//不为空就填写
                    edit.setText(readMessage);
                    edit.setSelection(readMessage.length());
                    Toast.makeText(this, "点击按钮读取数据成功："+readMessage, Toast.LENGTH_SHORT).show();//吐司提示更新成功
                }

                break;

        }
    }


    @Override
    protected void onDestroy() {
        super.onDestroy();
        String inputText = edit.getText().toString();
        save(inputText);//在点击手机的back退出后保存
    }
    //保存数据的函数
    public void save(String inputText) {
        FileOutputStream out = null;
        BufferedWriter writer = null;
        try {
            out = openFileOutput("data", Context.MODE_PRIVATE);//被保存的数据文件名为data
            writer = new BufferedWriter(new OutputStreamWriter(out));
            writer.write(inputText);
            Toast.makeText(this, "点击按钮保存数据成功："+inputText, Toast.LENGTH_SHORT).show();//吐司提示更新成功
        }
        //异常处理，为空不保存，打印异常信息
        catch (IOException e) {
            e.printStackTrace();
        } finally {
            try {
                if (writer != null) {
                    writer.close();
                }
            } catch (IOException e) {
                e.printStackTrace();
            }
        }
    }
    //读取文件
    public String load() {
        FileInputStream in = null;
        BufferedReader reader = null;
        StringBuilder content = new StringBuilder();
        try {
            in = openFileInput("data");//获取到了一个FileInputStream 对象
            //构建出了一个InputStreamReader 对象,使用InputStreamReader 构建出一个BufferedReader 对象
            reader = new BufferedReader(new InputStreamReader(in));
            String line = "";//通过BufferedReader 进行一行行地读取
            while ((line = reader.readLine()) != null) {
                content.append(line);
            }
        } catch (IOException e) {
            e.printStackTrace();
        } finally {
            if (reader != null) {
                try {
                    reader.close();
                } catch (IOException e) {
                    e.printStackTrace();
                }
            }
        }
        return content.toString();
    }
}
