void keyevent()
{
	int i = 0; int j = 0;

	char ch = _getch();
	switch (ch)
	{
	case 119:
	case 'w ':
	case 'W':

		{
			map[i - 1][j] += PLAYER;
			map[i][j] -=  PLAYER;

			}



		}
		break;
	case 97:
	case 'a ':
	case'A':

		{
			map[i][j - 1] += PLAYER;
			map[i][j] -=  PLAYER;



		}

			if (map[i][j - 2] == SPACE || map[i][j - 2] == DEST)

				map[i][j - 2] += BOX;

			}



		}
		break;
	case 115:
	case 's ':
	case'S':
		if (map[i + 1][j] == SPACE || map[i + 1][j] == DEST)
		{
			map[i + 1][j] += PLAYER;
			map[i][j] -= PLAYER;



		}
