求任意三位数之和

var num = prompt('请输入一个三位数：');
        var a = num%10;
        var b = parseInt((num%100)/10);
        var c = parseInt(num/100);
        var sum = a+b+c;
        alert('该三位数的个十百位数之和为：' + sum)

        for (var i = 1; i <= 9; i++) {
            for (var j = 1; j <= i; j++) {

        console.log(str);

打印倒三角形、正三角形

//倒三角形
        var str = '';
        for (var i = 1;i<=10;i++) {
            for (var j=i;j<=10;j++) {
                str = str + '*';
            }
            str += 'n';
        }
        console.log(str);
        //正三角形
        var str = '';
        for (var i = 1; i <= 10; i++) {
            for (var j = 1; j <= i; j++) {
                str = str + '*';
            }
            str += 'n';
        }
        console.log(str);
