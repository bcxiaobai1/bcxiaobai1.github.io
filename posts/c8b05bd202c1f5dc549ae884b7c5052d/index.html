一，游戏准备

我们先创建三个文件：<game.h><game.c><test.c>

game:h    用于存放游戏的函数

game.c  用于实现游戏  定义函数

test.c   用于存放游戏大致步骤

下棋：人下棋可以选择用* 电脑下棋应该用#，同时我们的棋盘大致是一个井字

综上：我们棋盘用的是字符型数组来打印  

提示：在test.c最开始加一个#include”game.h"这样才能使用game.h中的一些函数与头文件，

同时我们也可以把相关头文件全部放进game.h中  直接在test.c中加一个game.h就可以了

二，游戏思路

首先 用户见到的初始页面是菜单，所以我们一定要打印菜单，又因为游戏不是玩一把就结束了，用户还可以选择再来一把，因此应该是一个循环，而且还是一个do while循环（刚开始那个函数体必须先出来）菜单如下：

void menu() {
    printf("***********************n");
    printf("*******1.play**********n");
    printf("*******0.exit**********n");
    printf("***********************n");
}

出现菜单后，玩家选择，就应该有用户输入数字------->程序接收数字后，我们可以用switch语句进行选择是否进行游戏，接着就可以开始对游戏的实现进行编译了。**值得注意的是，这些都是在do while循环里面的哦

打印主体：

void game( ){}
int main() {
    int input = 0;
    do
    {
        
        menu();
        printf("请选择:>");（这里可以加一个玩法，是按照坐标的形式输入，不然有些玩家不会玩）
        scanf("%d", &input);
        switch (input) {
        case 1:game();
            break;
        case 2:printf("游戏结束！n");
            break;
        default:printf("请重新选择:>n");
            break;
        }
    } while (input);
    return 0;
}

三,游戏实现

在游戏准备时我们提到，应该是用字符型数组打印，所以我们应该在<game.h>中定义一下数组的行   列

 为什么要这样设计 不直接char [3][3]   这样设计是为了防止如果想要改变行和列的话 我们只需要在game.h上改动就是了。

初始话数组 ；char board[ROW][LINE] = {0};

初始化键盘：void Initbord(char board[ROW][LINE], int row,int line);（把数组全部初始成空格，有利于后续操作，后续只需要把我们打印的数字和空格交换就可以了）

void Initbord(char board[ROW][LINE], int row, int line)
{
    int i = 0;
    int j = 0;
    for (i = 0;i < row;i++) {
        for (j = 0;j < line;j++) {
            board[i][j] = ' ';
        }
    }
}

打印键盘：void Displayborad(char board[ROW][LINE], int row, int line)

玩家下棋：void Player_move(char board[ROW][LINE], int row, int line)

玩家下棋的是*；

假设我们输入的是x，y      在编写玩家下棋代码之前我们要思考一下，首先，我们的坐标必须在我们的行列范围之内，其次，玩家输入的坐标不会是从0开始的，都是从1开始的，所以如果我们输入的x，y必须减一才可以，然后我们下的棋应该是空格，这样想的话，代码便出来了。

void Player_move(char board[ROW][LINE], int row, int line)
{
    printf("玩家输入:>n");
    int x = 0;
    int y = 0;
    while (1) {
        scanf("%d%d", &x, &y);
        if (x > 0 && x <= row && y > 0 && y <= line) {
            if (board[x - 1][y - 1] == ' ') {
                board[x - 1][y - 1] = '*';
                break;
            }
            else {
                printf("该坐标已被占用，请重新输入n");
            }
        }
        else {
            printf("该数字无效，请重新输入有效数字n");
        }
    }
}
在玩家输入后我们应该再次打印棋盘，可以让玩家看到下棋后的棋盘，因此我们需要在test.c文件中再加一个    Player_move(board,ROW,LINE);

#define _CRT_SECURE_NO_WARNINGS 1
#include"game.h"
void menu() {
	printf("***********************n");
	printf("*******1.play**********n");
	printf("*******0.exit**********n");
	printf("***********************n");
}
void game( )
{
	char board[ROW][LINE] = {0};
	char ret = 0;
	Initbord(board, ROW, LINE);
	Displayborad(board,ROW,LINE);
	while (1) {
		Player_move(board, ROW, LINE);
		Displayborad(board, ROW, LINE);
	ret =	is_win(board, ROW, LINE);
	if (ret != 'c') {
		if (ret == '*') {
			printf("恭喜你 成功拿捏！n");
			break;
		}
		if (ret == '#') {
			printf("很遗憾，你输了n");
			break;
		}
		if (ret == 'q') {
			printf("技术很不错，平局哦！n");
			break;
		}
	}
		Computer_move(board, ROW, LINE);
		Displayborad(board, ROW, LINE);
		ret = is_win(board, ROW, LINE);

		if (ret != 'c') {
			if (ret == '*') {
				printf("恭喜你 成功拿捏！n");
				break;
			}
			if (ret == '#') {
				printf("很遗憾，你输了n");
				break;
			}
			if (ret == 'q') {
				printf("技术很不错，平局哦！n");
				break;
			}
		}

	}

}
int main() {
	int input = 0;
	do
	{
		srand((unsigned)time(NULL));
