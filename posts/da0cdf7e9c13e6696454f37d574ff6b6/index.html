# 设置窗口

    '月收入分析': (100, 300),
    '每日接待车辆': (300, 300),
    '车位利用率': (500, 300)
}

# 读取Excel文件
df = pd.read_excel("停车场信息表.xlsx", engine='openpyxl')
df['timein'] = pd.to_datetime(df['timein'])
df['timeout'] = pd.to_datetime(df['timeout'])
df['parking_duration'] = (df['timeout'] - df['timein']).dt.total_seconds() / 3600
