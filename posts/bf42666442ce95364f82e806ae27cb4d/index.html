function generateSolution() {

    const solution = Array.from({ length: 9 }, () => new Array(9).fill(0));



    function isSafe(row, col, num) {

        for (let i = 0; i < 9; i++) {

            if (solution[row][i] === num || solution[i][col] === num) return false;

            if (

                solution[3 * Math.floor(row / 3) + Math.floor(i / 3)][

                    3 * Math.floor(col / 3) + (i % 3)

                ] === num

            )

                return false;

        }

        return true;

    }



    function solve(row, col, startNum) {

        if (row === 9) return true;

        if (col === 9) return solve(row + 1, 0, startNum);

        if (solution[row][col] !== 0) return solve(row, col + 1, startNum);



        for (let numOffset = 0; numOffset < 9; numOffset++) {

            const num = ((startNum + numOffset) % 9) + 1;

            if (isSafe(row, col, num)) {

                solution[row][col] = num;

                if (solve(row, col + 1, startNum)) return true;

                solution[row][col] = 0;

            }

        }

        return false;

    }



    const startNum = Math.floor(Math.random() * 9) + 1;

    solve(0, 0, startNum);

    return solution;

}

在solve函数中，我们添加了一个名为startNum的参数，它是一个随机起始数字。我们还修改了循环中的数字选择逻辑，使其从startNum开始并随机调整数字的顺序。

function removeNumbers(solution, difficulty) {

    let numToRemove;

    if (difficulty === "easy") {

        numToRemove = 20;

    } else if (difficulty === "medium") {

        numToRemove = 35;

    } else if (difficulty === "hard") {

        numToRemove = 50;

    } else {

        numToRemove = 64; // 保留17个数字

    }



    while (numToRemove > 0) {

        const row = Math.floor(Math.random() * 9);

        const col = Math.floor(Math.random() * 9);

        if (solution[row][col] !== 0) {

            solution[row][col] = 0;

            numToRemove--;

        }

    }

    return solution;

}
