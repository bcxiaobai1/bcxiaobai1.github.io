 GM(1,1)所有计算及检验计算程序

x0=[71.1 72.4 72.4 72.1 71.4 72.0 71.6];
n=length(x0);
lamuda=x0(1:n-1)./x0(2:n);
x1=cumsum(x0);
z=(x1(1:n-1)+x1(2:n))/2;
B=[-z',ones(n-1,1)];Y=(x0(2:n))';
u=inv(B'*B)*B'*Y;
x1(1)=x0(1);
x2(1)=x0(1);
xy(1)=x0(1);
for k=1:n-1
    x2(k+1)=(x0(1)-u(2)/u(1))*exp(-u(1)*k)+u(2)/u(1);
end
xy=[x0(1),diff(x2)];
err=x0-xy;
errb=err./x0;
a=u(1);
rho=(1-(1-0.5*a)/(1+0.5*a))*lamuda;
[x0',x1',[0,z]',xy',err',errb',[0,rho]']
