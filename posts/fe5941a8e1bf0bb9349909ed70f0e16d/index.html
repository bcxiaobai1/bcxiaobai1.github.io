题解

BFS遍历，用’1‘记录走过的点和不能走的点，将每个点代表的二维矩阵 n ∗ m {n*m} n∗m坐标化为唯一的一维坐标，即 k = ( i , j ) = ( i − 1 ) ∗ m + j {k=(i,j)=(i-1)*m+j} k=(i,j)=(i−1)∗m+j，以此来记录每个点的前驱pre[k]，和方向答案ans[k]

AC代码

#include<bits/stdc++.h>
using namespace std;

char g[50][90];
typedef pair<int,int> PII;
char c[4]={'D','L','R','U'};
int h[4]={1,0,0,-1},l[4]={0,-1,1,0};
int pre[2000];
char ans[2000];
int n,m;

void bfs(){
	queue<PII> q;
	PII k;
	k.first=1,k.second=1;
	q.push(k);
	pre[1]=1;g[1][1]='1';
	
	while(!q.empty()){
		k=q.front();q.pop();
		//cout<<" "<<k.first<<" "<<k.second<<" "<<pre[(k.first-1)*m+k.second]<<" "<<ans[(k.first-1)*m+k.second]<<endl;
		for(int i=0;i<4;i++){
			int x=k.first+h[i],y=k.second+l[i];
			if(x>0&&y>0&&x<=n&&y<=m&&g[x][y]=='0'){
				g[x][y]='1';
				ans[(x-1)*m+y]=c[i];
				q.push({x,y});
				if(pre[(x-1)*m+y]==0)pre[(x-1)*m+y]=(k.first-1)*m+k.second;
			}
		}
	}	
}

void prin(int x){//输出
	if(x==pre[x]) return;
	prin(pre[x]);
	cout<<ans[x];
}

int main(){
	//cout<<"DDDDRRURRRRRRDRRRRDDDLDDRDDDDDDDDDDDDRDDRRRURRUURRDDDDRDRRRRRRDRRURRDDDRRRRUURUUUUUUULULLUUUURRRRUULLLUUUULLUUULUURRURRURURRRDDRRRRRDDRRDDLLLDDRRDDRDDLDDDLLDDLLLDLDDDLDDRRRRRRRRRDDDDDDRR";
	cin>>n>>m;
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++) cin>>g[i][j];
	
	bfs();
	prin(n*m);
  return 0;
}

给定一棵包含 N 个节点的完全二叉树，树上每个节点都有一个权值，按从

权值之和最大？如果有多个深度的权值和同为最大，请你输出其中最小的深度。

		num=0;

		if(v>=n) break;

等差数列（20分）（贪心）

Description

数学老师给小明出了一道等差数列求和的题目。但是粗心的小明忘记了一部分的数列，只记得其中 N 个整数。现在给出这 N 个整数，小明想知道包含这 N 个整数的最短的等差数列有几项？

Input
输入的第一行包含一个整数 N。

第二行包含 N 个整数 A 1 ,A 2 ,··· ,A N 。(注意 A 1 ∼ A N 并不一定是按等差数列中的顺序给出)

对于所有评测用例，2 ≤ N ≤ 100000，0 ≤ A i ≤ 10^9 。

Output
输出一个整数表示答案。

Sample Input 1

5
2 6 4 10 20
Sample Output 1

10

题解

找到两两之间最小的差值，计算最小数和最大数之间有多少个等差项

AC

#include<bits/stdc++.h>
using namespace std;

int n;
int a[100086];
 
int main(){
	cin>>n;
	for(int i=1;i<=n;i++) cin>>a[i];
	
	sort(a+1,a+1+n);
	int minn=0x3f3f3f3f;
	for(int i=1;i<n;i++) minn=min(minn,a[i+1]-a[i]);
	if(minn==0){
		cout<<n;return 0;
	}
	int ans=a[n]-a[1];
	ans/=minn;ans++;
	cout<<ans;
  return 0;
}

后缀表达式（25分）（二叉树性质+贪心）

Description

给定 N 个加号、M 个减号以及 N + M + 1 个整数 A 1 ,A 2 ,··· ,A N+M+1 ，小

明想知道在所有由这 N 个加号、M 个减号以及 N + M +1 个整数凑出的合法的

后缀表达式中，结果最大的是哪一个？

请你输出这个最大的结果。

例如使用1 2 3 + -，则 “2 3 + 1 -” 这个后缀表达式结果是 4，是最大的。

Input
第一行包含两个整数 N 和 M。

第二行包含 N + M + 1 个整数 A 1 ,A 2 ,··· ,A N+M+1 。

Output
输出一个整数，代表答案。

Sample Input 1

1 1
1 2 3
Sample Output 1

4
Hint

0 ≤ N, M ≤ 100000，−10^9 ≤ A i ≤ 10^9 。

题解

其实后缀表达式已经提醒我们很多了，，，
首先后缀表达式是什么？
你按照二叉树前中后遍历的出来的式子就是前中后缀表达式。
一般来说，一棵树需要中+前或后缀才能表达出来，这里需要的是告诉我们叶子结点（数值）和根节点（符号）要我们自己来构造最后值最大的后缀表达式；
我吗可以发现，一个有m个减号的关系式，可以构造出 1 — n + m {1—n+m} 1—n+m个有减号，也就是说，除了第一个数没有符号以及最后一个数必须要是减号，任意一个表达式中间的数都可以我们随意加减号。
那么要使表达式最大，我们第一个数就是最大数，最后一个数就是减去最小数，中间的数如果大于0则不变号，小于0则变号
特别注意的是，当 m = 0 {m=0} m=0时，我们不能改变任何符号，只能输出原来序列相加值。

AC

#include<bits/stdc++.h>
using namespace std;

int n,m;
int a[200086];

int main(){
    cin>>n>>m;
    int k=1+m+n;
    for(int i=0;i<k;i++) cin>>a[i];
    
    sort(a,a+k);
    long long ans=0;
    if(!m){
        for(int i=0;i<k;i++) ans+=a[i];
    }else{
        ans=a[k-1]-a[0];
        for(int i=1;i<k-1;i++) ans+=abs(a[i]);
    }
    cout<<ans;
    return 0;
}

在游戏《星际争霸 II》中，高阶圣堂武士作为星灵的重要 AOE 单位，在游戏的中后期发挥着重要的作用，其技能”灵能风暴“可以消耗大量的灵能对一片区域内的敌军造成毁灭性的伤害。

经常用于对抗人类的生化部队和虫族的刺蛇飞龙等低血量单位。

你控制着 n 名高阶圣堂武士，方便起见标为 1,2,⋅⋅⋅,n。

每名高阶圣堂武士需要一定的灵能来战斗，每个人有一个灵能值 ai 表示其拥有的灵能的多少（ai 非负表示这名高阶圣堂武士比在最佳状态下多余了 ai 点灵能，ai 为负则表示这名高阶圣堂武士还需要 −ai 点灵能才能到达最佳战斗状态）。

现在系统赋予了你的高阶圣堂武士一个能力，传递灵能，每次你可以选择一个 i∈[2,n−1]，若 ai≥0 则其两旁的高阶圣堂武士，也就是 i−1、i+1 这两名高阶圣堂武士会从 i 这名高阶圣堂武士这里各抽取 ai 点灵能；若 ai<0 则其两旁的高阶圣堂武士，也就是 i−1,i+1 这两名高阶圣堂武士会给 i 这名高阶圣堂武士 −ai 点灵能。

形式化来讲就是 ai−1+=ai,ai+1+=ai,ai−=2ai。

灵能是非常高效的作战工具，同时也非常危险且不稳定，一位高阶圣堂武士拥有的灵能过多或者过少都不好，定义一组高阶圣堂武士的不稳定度为 maxni=1|ai|，请你通过不限次数的传递灵能操作使得你控制的这一组高阶圣堂武士的不稳定度最小。

输入格式

接下来依次输入每一组询问。

每组询问的第一行包含一个正整数 n，表示高阶圣堂武士的数量。

接下来一行包含 n 个数 a 1 , a 2 , ⋅ ⋅ ⋅ , a n {a1,a2,⋅⋅⋅,an} a1,a2,⋅⋅⋅,an。

输出格式
输出 T 行。

每行一个整数依次表示每组询问的答案。

数据范围
1 ≤ T ≤ 3 , 3 ≤ n ≤ 300000 , ∣ a i ∣ ≤ 1 0 9 {1≤T≤3,3≤n≤300000,|ai|≤10^9} 1≤T≤3,3≤n≤300000,∣ai∣≤109,

AC

#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstring>
using namespace std;

const int N=3e5+10;

//由于a[]可能达到1e9所以需要使用到LL
typedef long long LL;

LL a[N];    //用于存放初始灵能值
LL s[N];    //用于存放前缀和
LL f[N];    //用于存放最终的有序序列

bool st[N];

int main()
{
    int T;
    scanf("%d",&T);

    while (T--)
    {
        int n;
        scanf("%d",&n);

        s[0]=0;// 注意这一步不要忘了

        for(int i=1;i<=n;i++)
            scanf("%lld",&a[i]),s[i]=s[i-1]+a[i];


        LL s0=s[0],sn=s[n];

        if(s0>sn)
            swap(s0,sn);

        sort(s,s+1+n);

        //找到排序后 s0,sn的位置
        for(int i=0;i<=n;i++)
            if(s0==s[i])
            {
                s0=i;
                break;
            }

        for(int i=0;i<=n;i++)
            if(sn==s[i])
            {
                sn=i;
                break;
            }

        memset(st,false,sizeof st);

        //构造重叠部分最小的序列

        int l=0,r=n;

        for(int i=s0;i>=0;i-=2)
            f[l++]=s[i],st[i]=true;

        for(int i=sn;i<=n;i+=2)
            f[r--]=s[i],st[i]=true;

        for(int i=0;i<=n;i++)
            if(st[i]==false)
                f[l++]=s[i];

        LL res=0;

        for(int i=1;i<=n;i++)
            res=max(res,abs(f[i]-f[i-1]));

        printf("%lldn",res);

    }

    return 0;
}
